EXERCÍCIO 48






Algoritmo "ex48"
Var
peso, altura, imc: Real
Inicio
Escreva("Digite seu peso:")
Leia(peso)
Escreva("Digite sua altura (em centímetros):")
Leia(altura)
imc <- peso / (altura * altura)
imc <- imc * 10000
EscrevaL("SEU IMC: ", imc)
Se imc < 18 Entao
EscrevaL(" ABAIXO DO PESO.")
Senao
Se imc < 25 Entao
EscrevaL(" NORMAL.")
Senao
EscrevaL(" ACIMA DO PESO.")
FimSe
FimSe
FimAlgoritmo
